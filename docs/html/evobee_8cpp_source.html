311 void processConfigOptions(int argc, char **argv)
312 {
313  try
314  {
315  std::string config_file;
316 
317  // Declare a group of options that will be
318  // allowed only on command line

320  generic.add_options()

322  ("help,h", "display this help message")
323  ("config,c", po::value<std::string>(&config_file)->default_value("evobee.cfg.json"), "configuration file")
324  ("quiet,q", "disable verbose progress messages on stdout");
325 
326  po::options_description cmdline_options;
327  cmdline_options.add(generic);
328 
329  po::options_description visible("Allowed options");
330  visible.add(generic);
331 
332  po::positional_options_description p;
333 
334  po::variables_map vm;
335  store(po::command_line_parser(argc, argv).options(cmdline_options).positional(p).run(), vm);
336  notify(vm);
337 
338  // first deal with command line options that do not involve running the model
339  if (vm.count("help"))
340  {
341  std::cout << visible << std::endl;
342  exit(0);
343  }
344 
345  if (vm.count("version"))
346  {
347  std::cout << "EvoBee version " << evobee_VERSION_MAJOR << "." << evobee_VERSION_MINOR << "."
348  << evobee_VERSION_PATCH << "." << evobee_VERSION_TWEAK << std::endl;
349  exit(0);
350  }
351 
352  // get the configuration file name
353  std::ifstream ifs(config_file.c_str());
354  if (!ifs)
355  {
356  std::cerr << "Unable to open config file: " << config_file << std::endl;
357  exit(1);
358  }
359 
360  // is the command line explicity asking us to suppress stdout notifications?
361  if (vm.count("quiet"))
362  {
363  ModelParams::setVerbose(false);
364  ModelParams::setCommandLineQuiet(true);
365  }
366 
367  // process the contents of the configuration file
368  processJsonFile(ifs);
369 
370  // if command line explicitly asked for no notifications, then set that option
371  // again now, in case the configuration file conflicted with this (the command line
372  // flag should override what is in the config file for this)
373  if (ModelParams::commandLineQuiet())
374  {
375  ModelParams::setVerbose(false);
376  }
377 
378  }
379  catch (std::exception &e)
380  {
381  std::cerr << e.what() << std::endl;
382  exit(0);
383  }
384 
385  ModelParams::setInitialised();
