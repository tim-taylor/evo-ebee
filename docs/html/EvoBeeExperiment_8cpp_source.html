85 void EvoBeeExperiment::runStandardExperiment()
86 {
87  for (int gen = 0; gen < ModelParams::getSimTerminationNumGens(); ++gen)
88  {
90  // REPRODUCTION PHASE //
92 
93  // if this is not the first generation, we need to reinitialise
94  // all plants and pollinators based upon which plants got pollinated
95  // in the previous generation. These details will be logged if
96  // necessary. Also reset the visualisation if needed.
97  if (gen > 0)
98  {
99  m_Model.initialiseNewGeneration();
100  if (m_bVis)
101  {
102  m_Visualiser.reset();
103  }
104  }
105 
107  // FORAGING PHASE //
109 
110  bool endOfGen = false;
111  bool bContinue = true;
112  int step = 0;
113 
114  do
115  {
116  // perform the next simulation step
117  m_Model.step();
118 
119  // perform logging of pollinators if required
120  if (ModelParams::logging() && (step % m_iLogUpdatePeriod == 0))
121  {
122  if (ModelParams::logPollinatorsIntraPhaseFull()) {
123  callLoggerMethod(&Logger::logPollinatorsIntraPhaseFull);
124  }
125  if (ModelParams::logFlowersIntraPhaseFull()) {
126  callLoggerMethod(&Logger::logFlowersIntraPhaseFull);
127  }
128  if (ModelParams::logFlowersIntraPhaseSummary()) {
129  callLoggerMethod(&Logger::logFlowersIntraPhaseSummary);
130  }
131  }
132 
133  // perform visualisation if required
134  if ((m_bVis) && (step % m_iVisUpdatePeriod == 0))
135  {
136  bContinue = m_Visualiser.update();
137  if (!bContinue)
138  {
139  break; // may want to ask the user for confirmation first?
140  }
141  }
142 
143  // this step is now finished, so ...
144  // ... advance step count
145  ++step;
146  // ... and check whether the current generation is now complete
147  switch (ModelParams::getGenTerminationType())
148  {
149  case GenTerminationType::NUM_SIM_STEPS:
150  {
151  if (step >= ModelParams::getGenTerminationIntParam())
152  {
153  endOfGen = true;
154  }
155  break;
156  }
157  case GenTerminationType::NUM_POLLINATOR_STEPS:
158  {
160  break;
161  }
162  case GenTerminationType::POLLINATED_FRACTION_ALL:
163  {
164  if (m_Model.getEnv().getPollinatedFracAll() >= ModelParams::getGenTerminationFloatParam())
165  {
166  endOfGen = true;
167  }
168  break;
169  }
170  case GenTerminationType::POLLINATED_FRACTION_SPECIES1:
171  {
172  if (m_Model.getEnv().getPollinatedFracSpecies1() >= ModelParams::getGenTerminationFloatParam())
173  {
174  endOfGen = true;
175  }
176  break;
177  }
178  default:
179  {
180  throw std::runtime_error("Unexpected generation-termination-type encountered!");
181  }
182  }
183  }
184  while (!endOfGen); // end of foraging phase
185 
187  // LOGGING AT END OF FORAGING PHASE //
189 
190  // perform logging of flowers at the end of the generation if required
191  if (ModelParams::logging())
192  {
193  if (ModelParams::logFlowersInterPhaseFull()) {
194  callLoggerMethod(&Logger::logFlowersInterPhaseFull);
195  }
196  if (ModelParams::logFlowersInterPhaseSummary()) {
197  callLoggerMethod(&Logger::logFlowersInterPhaseSummary);
198  }
199  if (ModelParams::logFlowerMPsInterPhaseSummary()) {
200  callLoggerMethod(&Logger::logFlowerMPsInterPhaseSummary);
201  }
202  if (ModelParams::logFlowerInfoInterPhaseSummary()) {
203  callLoggerMethod(&Logger::logFlowerInfoInterPhaseSummary);
204  }
205  if (ModelParams::logPollinatorsInterPhaseFull()) {
206  callLoggerMethod(&Logger::logPollinatorsInterPhaseFull);
207  }
208  if (ModelParams::logPollinatorsInterPhaseSummary()) {
209  callLoggerMethod(&Logger::logPollinatorsInterPhaseSummary);
210  }
211  }
212 
213  if (!bContinue)
214  {
215  break;
216  }
217  }
218 
220  // FINAL TIDY UP AT END OF RUN //
222 
223  // at end of run, transfer all log files to the final destitination directory
224  // if one has been specified
225  if (ModelParams::logging())
226  {
227  m_Logger.transferFilesToFinalDir();
228  }
