224 // Attempt to transfer the requested number of Pollen grains from the specified
225 // pollinator's store to the flower's stigma. If any of the deposited pollen
226 // is from the same species, the flower will be pollinated. If less pollen gets
227 // transferred to the flower than the requested number, the difference between
228 // the requested and actual number of pollen grains transferred is removed
229 // from the pollinator's store and treated as lost.
230 //
231 int Flower::transferPollenFromPollinator(PollenVector& pollinatorStore, int suggestedNum)
232 {
233  // first figure out how many grains we should attempt to transfer,
234  // taking into account the suggested number from the pollinator, the
235  // remaining capacity of the stigma, and the amount of pollen actually
236  // being carried by the pollinator
237  int attemptedNum = std::min({suggestedNum,
238  m_iStigmaMaxPollenCapacity - (int)m_StigmaPollen.size(),
239  (int)pollinatorStore.size()});
240 
241  int actualNum = 0;
242 
243  // now calculate how many grains we can actually move from the pollinatorStore,
244  // taking account of the pollen species if necessary, and ensure that all grains
245  // to be moved are placed at the end of the pollinatorStore ready for being
246  // moved in bulk
247  if (FloweringPlant::cloggingAll())
248  {
249  // No restriction on pollen species, so we can grab any pollen from
250  // the pollinator. Note that the pollinator's store should already be shuffled,
251  // so we can just pick the required number of grains from the end of the
252  // store, and this will be a random selection
253  actualNum = attemptedNum;
254  }
255  else
256  {
257  // We can only transfer pollen of the same species as this flower, so we need
258  // to look through the pollinator's store and select which grains we will take.
259  // For efficiency, we do this by first searching through the pollinator's store
260  // and moving any matching pollen to the end of the store (we exit this process
261  // when we've found enough pollen, or when we've reached the end of the store).
262  if (!pollinatorStore.empty())
263  {
264  auto nextItr = pollinatorStore.begin();
265  auto swapItr = pollinatorStore.end();
266 
267  while ((nextItr != swapItr) && (actualNum < attemptedNum))
268  {
269  if (FloweringPlant::pollenTransferToStigmaAllowed(nextItr->pSource, this))
270  {
271  --swapItr;
272  if (nextItr != swapItr)
273  {
274  std::iter_swap(nextItr, swapItr);
275  }
276  ++actualNum;
277  }
278  else
279  {
280  ++nextItr;
281  }
282  }
283  }
284  }
285 
286  // Whether m_bPollenClogging is true or false, we have now figured out how many grains
287  // are to be moved, and the grains we want to move are now at the end of the pollinatorStore.
288  // Now we actually move those grains to the stigma store in one operation, and delete
289  // them from the pollinatorStore.
290  if (actualNum > 0)
291  {
292  std::move(pollinatorStore.end()-actualNum,
293  pollinatorStore.end(),
294  std::back_inserter(m_StigmaPollen));
295 
296  pollinatorStore.erase(pollinatorStore.end()-actualNum, pollinatorStore.end());
297 
298  // if not already pollinated, check whether that has now changed!
299  if (!m_bPollinated)
300  {
301  if (FloweringPlant::cloggingNone())
302  {
303  // in the case of no pollen clogging, only pollen from this species is allowed
304  // on the stigma, so we know for sure that the flower is now pollinated
305  m_bPollinated = true;
306  }
307  else
308  {
309  // pollen could be of any species, so we need to see whether any of
310  // the new pollen is actually from the same species as this flower
311  auto it = std::find_if( m_StigmaPollen.end()-actualNum,
312  m_StigmaPollen.end(),
313  [this](Pollen& p){return (p.speciesId == this->m_SpeciesId);} );
314  m_bPollinated = (it != m_StigmaPollen.end());
315  }
316 
317  if (m_bPollinated)
318  {
319  m_pPlant->setPollinated();
320  }
321  }
322  }
323 
324  // Finally, compare the number of grains actually deposited on the stigma with the amount
325  // of the requested transfer. If the actual amount is less, then the pollinator just
326  // loses the difference anyway
327  if (actualNum < suggestedNum)
328  {
329  // first ensure we don't try to lose more than is actually in the pollinator's store
330  int numToLose = std::min({ suggestedNum-actualNum, (int)pollinatorStore.size() });
331  pollinatorStore.erase(pollinatorStore.end()-numToLose, pollinatorStore.end());
332  }
333 
334  return actualNum;
335 }
