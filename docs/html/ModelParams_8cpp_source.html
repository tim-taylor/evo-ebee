459 void ModelParams::addPlantTypeConfig(PlantTypeConfig& pt)
460 {
461  // first do some sanity checking of values
462  if (pt.flowerMPInitMin > pt.flowerMPInitMax)
463  {
464  std::stringstream msg;
465  msg << "PlantTypeConfig error for type " << pt.species
466  << ": flowerMPInitMin (" << pt.flowerMPInitMin
467  << ") cannot be larger than flowerMPInitMax (" << pt.flowerMPInitMax << ")!";
468  throw std::runtime_error(msg.str());
469  }
470 
471  if (pt.diffMPIsDiffSpecies && (pt.flowerMPInitMin != pt.flowerMPInitMax))
472  {
473  // deal with case where we need to create a separate species for each
474  // different marker point that might be generated
475  for (MarkerPoint mp = pt.flowerMPInitMin; mp <= pt.flowerMPInitMax; mp+=10)
476  {
477  PlantTypeConfig newpt{pt};
478  std::stringstream species;
479  species << "PlantSpecies" << mp;
480  newpt.species = species.str();
481  newpt.flowerMPInitMin = mp;
482  newpt.flowerMPInitMax = mp;
483  newpt.diffMPIsDiffSpecies = false;
484  m_PlantTypes.push_back(newpt);
485  FloweringPlant::registerSpecies(newpt.species);
486  m_bSyntheticRegularMarkerPointsAdded = true; // we'll check whether this is okay in ModelParams::checkConsistency
487  }
488  }
489  else
490  {
491  // this is the straightforward default case where we just need to record a
492  // single new plant type
493  m_PlantTypes.push_back(pt);
494  FloweringPlant::registerSpecies(pt.species);
495  }

635 // If the config parameter auto-distribs is set to true, we now have to automatically generate
636 // the required set of PlantTypeDistributionConfig objects here based upon the specification
637 // provided by the related config parameters.
638 void ModelParams::autoGeneratePtds()
639 {
640  assert(m_bPtdAutoDistribs);
641 
642  // First we initialise an 2D array (actually stored in a 1D vector) which contains
643  // info about which species will go into which patch. This initialisation is done by
644  // the helper method initialiseAutoGenPtdSpeciesPatchMap(), which takes into account
645  // whether the param m_bPtdAutoDistribRegular is true or false, to determine
646  // whether distribution is regular (checkerboard) or stochastic. Having created the
647  // map, within the nested loop below we can just call another helper method
648  // getAutoGenPtdSpeciesForPatch() to extract the corresponding species from the
649  // array for any given x and y patch coordinate.
650  std::vector<const std::string*> speciesPatchMap; // for each patch, record name of inhabitant species
651  initialiseAutoGenPtdSpeciesPatchMap(speciesPatchMap);
652 
653  int areaWidth = m_iEnvSizeX / m_iPtdAutoDistribNumCols;
654  int areaHeight = m_iEnvSizeY / m_iPtdAutoDistribNumRows;
655  int margin = (int)(0.5 * (float)std::min(areaWidth, areaHeight) * m_fPtdAutoDistribAreaMargin);
656 
657  int curX = 0;
658  int curY = 0;
659 
660  for (int x = 0; x < m_iPtdAutoDistribNumCols; x++)
661  {
662  for (int y = 0; y < m_iPtdAutoDistribNumRows; y++)
663  {
664  PlantTypeDistributionConfig ptdc;
665  ptdc.species = getAutoGenPtdSpeciesForPatch(x, y, speciesPatchMap);
666 
667  ptdc.areaTopLeft.x = curX + margin;
668  ptdc.areaTopLeft.y = curY + margin;
669  ptdc.areaBottomRight.x = curX + areaWidth - 1 - margin;
670  ptdc.areaBottomRight.y = curY + areaHeight - 1 - margin;
671 
672  if (ptdc.species == ModelParams::m_strNoSpecies)
673  {
674  // this is an area that will initially be empty (because m_bPtdAutoDistribEqualNums==true means
675  // that we may have a few empty areas). So in this case, we give the area an initial density of
676  // zero, and arbitrarily assign it to the first plant species in the map to keep the rest of
677  // the code happy
678  ptdc.density = 0.0;
679  ptdc.species = FloweringPlant::getSpeciesMap().begin()->second;
680  }
681  else
682  {
683  ptdc.density = m_fPtdAutoDistribDensity;
684  }
685 
686  ptdc.refuge = false;
687  ptdc.refugeAlienInflowProb = 0.0; // not required if refuge = false
688  ptdc.seedOutflowAllowed = m_bPtdAutoDistribSeedOutflowAllowed;
689  ptdc.seedOutflowRestricted = false; // not required if seedOutflowAllowed = false
690  ptdc.seedOutflowProb = 1.0; // not required if seedOutflowRestricted = false
691  ptdc.reproLocalDensityConstrained = true;
692  ptdc.reproLocalDensityMax = m_fPtdAutoDistribDensity;
693 
694  addPlantTypeDistributionConfig(ptdc);
695 
696  curY += areaHeight;

705 // private helper method for ModelParams::autoGeneratePtds()
706 void ModelParams::initialiseAutoGenPtdSpeciesPatchMap(std::vector<const std::string*>& speciesPatchMap)
707 {
708  assert(speciesPatchMap.empty());
709 
710  int numPatches = m_iPtdAutoDistribNumCols * m_iPtdAutoDistribNumRows;
711 
712  speciesPatchMap.reserve(numPatches);
713 
714  const std::map<unsigned int, std::string>& flowerSpeciesMap = FloweringPlant::getSpeciesMap();
715 
716  int numSpecies = flowerSpeciesMap.size();
717  int numPatchesToFill = m_bPtdAutoDistribEqualNums ? numPatches - (numPatches % numSpecies) : numPatches;
718  int numPatchesFilled = 0;
719 
720  if (m_bPtdAutoDistribRegular)
721  {
722  // allocate species to patches in a regular checkerboard pattern
723 
724  // NB This algorithm does not work well if numSpecies > m_iPtdAutoDistribNumRows/Cols!!
725  assert(numSpecies <= m_iPtdAutoDistribNumRows);
726  assert(numSpecies <= m_iPtdAutoDistribNumCols);
727 
728  auto nextColFirstSpeciesItr = flowerSpeciesMap.begin();
729  for (int x = 0; x < m_iPtdAutoDistribNumCols; x++)
730  {
731  auto nextSpeciesItr = nextColFirstSpeciesItr;
732  for (int y = 0; y < m_iPtdAutoDistribNumRows; y++)
733  {
734  if (numPatchesFilled < numPatchesToFill) {
735  speciesPatchMap.push_back(&(nextSpeciesItr->second));
736  numPatchesFilled++;
737  nextSpeciesItr++;
738  if (nextSpeciesItr == flowerSpeciesMap.end())
739  {
740  nextSpeciesItr = flowerSpeciesMap.begin();
741  }
742  }
743  else {
744  speciesPatchMap.push_back(&ModelParams::m_strNoSpecies);
745  }
746  }
747  nextColFirstSpeciesItr++;
748  if (nextColFirstSpeciesItr == flowerSpeciesMap.end())
749  {
750  nextColFirstSpeciesItr = flowerSpeciesMap.begin();
751  }
752  }
753  }
754  else {
755  // species are to be distributed stochastically across the environment
756 
757  // first allocate species to patches in a regular checkerboard pattern
758  auto nextSpeciesItr = flowerSpeciesMap.begin();
759  for (int x = 0; x < m_iPtdAutoDistribNumCols; x++)
760  {
761  for (int y = 0; y < m_iPtdAutoDistribNumRows; y++)
762  {
763  if (numPatchesFilled < numPatchesToFill) {
764  speciesPatchMap.push_back(&(nextSpeciesItr->second));
765  numPatchesFilled++;
766  nextSpeciesItr++;
767  if (nextSpeciesItr == flowerSpeciesMap.end()) {
768  nextSpeciesItr = flowerSpeciesMap.begin();
769  }
770  }
771  else {
772  speciesPatchMap.push_back(&ModelParams::m_strNoSpecies);
773  }
774  }
775  }
776 
777  // now just jumble everything up!
778  std::shuffle(speciesPatchMap.begin(), speciesPatchMap.end(), EvoBeeModel::m_sRngEngine);
779  }
780 }

812 // check consistency of parameters once the whole config file has been processed
813 void ModelParams::checkConsistency()
814 {
815  switch (m_GenTerminationType)
816  {
817  case GenTerminationType::NUM_SIM_STEPS:
818  {
819  if (m_iGenTerminationParam < 0)
820  {
821  throw std::runtime_error("Must provide an int parameter for termination type num-sim-steps");
822  }
823  break;
824  }
825  case GenTerminationType::NUM_POLLINATOR_STEPS:
826  {
827  if (m_iGenTerminationParam < 0)
828  {
829  throw std::runtime_error("Must provide an int parameter for termination type num-pollinator-steps");
830  }
831  break;
832  }
833  case GenTerminationType::POLLINATED_FRACTION_ALL:
834  {
835  if (m_fGenTerminationParam <= EvoBee::FLOAT_COMPARISON_EPSILON || m_fGenTerminationParam > 1.0 + EvoBee::FLOAT_COMPARISON_EPSILON)
836  {
837  throw std::runtime_error("Must provide a float parameter in range [0.0,1.0) for termination type pollinated-fraction-all");
838  }
839  break;
840  }
841  case GenTerminationType::POLLINATED_FRACTION_SPECIES1:
842  {
843  if (m_fGenTerminationParam <= EvoBee::FLOAT_COMPARISON_EPSILON || m_fGenTerminationParam > 1.0 + EvoBee::FLOAT_COMPARISON_EPSILON)
844  {
845  throw std::runtime_error("Must provide an float parameter in range [0.0,1.0) for termination type pollinated-fraction-species1");
846  }
847  break;
848  }
849  default:
850  {
851  throw std::runtime_error("Error: encountered unknown GenTerminationType in ModelParams::checkConsistency(). Aborting!");
852  }
853  }
854 
855  switch (m_ColourSystem) {
856  case ColourSystem::REGULAR_MARKER_POINTS:
857  {
858  for (auto &pc : m_PollinatorConfigs) {
859  if ((!pc.visDataEquallySpaced) || (pc.visDataMPStep < 1)) {
860  throw std::runtime_error("Error: with colour-system set to 'regular-marker-points' visual data must be defined at equally-spaced marker points");
861  }
862  }
863  break;
864  }
865  case ColourSystem::ARBITRARY_DOMINANT_WAVELENGTHS:
866  {
867  if (m_bSyntheticRegularMarkerPointsAdded) {
868  throw std::runtime_error("Error: with colour-system set to 'arbitrary-dominant-wavelengths' all PlantTypeConfigs must have flowerMPInitMin=flowerMPInitMax");
869  }
870 
871  for (auto& pc : m_PollinatorConfigs) {
872  pc.visDataMPStep = 1; // step has no meaning for ARBITRARY_DOMINANT_WAVELENGTHS, so we just ensure it is set to 1 here rather than some random other number
873  if (pc.learningStrategy != PollinatorLearningStrategy::STAY_INNATE) {
874  // At present the code has only been designed to cope with STAY_INNATE for ColourSystem::ARBITRARY_DOMINANT_WAVELENGTHS, so
875  // throw an error if this is not the case
876  throw std::runtime_error("Error: with colour-system set to 'arbitrary-dominant-wavelengths' all pollinator's learning strategies should be set to 'stay-innate'");
877  }
878  }
879 
880  auto& visData = getVisData();
881 
882  // Check that each entry in PollinatorConfig::visData matches a type of plant in the PlantTypeConfigs, and vice versa
883  bool allOK = true;
884  if (m_PlantTypes.size() != visData.size()) {
885  allOK = false;
886  }
887  else {
888  // go through plant types one by one checking that it has been successfully matched to an visdata entry
889  // (this should already be guaranteed by the action of ModelParams::pairPlantTypeConfigsToVisData())
890  for (auto& ptc : m_PlantTypes) {
891  if (ptc.flowerVisDataPtr == nullptr) {
892  allOK = false;
893  break;
894  }
895  }
896  }
897  if (!allOK) {
898  throw std::runtime_error("Error: each dominant wavelength associated with a PlantTypeConfig must match an entry in the vis-data array and vice versa");
899  }
900  break;
901  }
902  default:
903  {
904  throw std::runtime_error("Error: encountered unknown ColourSystem in ModelParams::checkConsistency(). Aborting!");
905  }
906  }
907 
909  // - are all pollinator and hive areas completely within the environment area?
910 }
911 
912 
913 // for each PlantTypeConfig, check that its flowerVisDataID matches an ID in VisData, and initialise a pointer in the
914 // PlantTypeConfig to point to the corresponding VisData entry. If no match is found for a PlantTypeConfig, throw an
915 // exception.
916 void ModelParams::pairPlantTypeConfigsToVisData()
917 {
918  // this method is only designed to be used with ColourSystem::ARBITRARY_DOMINANT_WAVELENGTHS
919  assert(m_ColourSystem == ColourSystem::ARBITRARY_DOMINANT_WAVELENGTHS);
920 
921  auto& visData = getVisData();
922 
923  for (auto& ptc : m_PlantTypes)
924  {
925  // first check our assumption that flower marker points have not been set explicitly in this case
926  assert(ptc.flowerMPInitMin == 0);
927  assert(ptc.flowerMPInitMax == 0);
928 
929  // and check that the vis-data-id has been set
930  assert(ptc.flowerVisDataID >= 0);
931 
932  int visDataID = ptc.flowerVisDataID;
933 
934  auto it = std::find_if(
935  visData.begin(),
936  visData.end(),
937  [visDataID](const VisualStimulusInfo& info) {return (visDataID == info.id);}
938  );
939 
940  if (it == visData.end()) {
941  std::stringstream msg;
942  msg << "In ModelParams::pairPlantTypeConfigsToVisData(): unable to find matching entry in vis-data\n"
943  << " for PlantTypeConfig with flower-vis-data-id = " << visDataID << "\n";
944  throw std::runtime_error(msg.str());
945  }
946 
947  ptc.flowerVisDataPtr = &(*it);
948  }
949 }
