484 // Set this Colour using HSV values
485 // Internally, this method converts the given HSV colour to RGB
486 // This code is from https://www.codespeedy.com/hsv-to-rgb-in-cpp/
487 void Colour::setHSV(float H, float S, float V)
488 {
489  if(H>360 || H<0 || S>100 || S<0 || V>100 || V<0) {
490  throw std::runtime_error("Colour::setHSV() parameters are out of range!");
491  }
492 
493  float s = S/100;
494  float v = V/100;
495  float C = s*v;
496  float X = C*(1-std::abs(std::fmod(H/60.0, 2)-1));
497  float m = v-C;
498  float r,g,b;
499 
500  if(H >= 0 && H < 60){
501  r = C,g = X,b = 0;
502  }
503  else if(H >= 60 && H < 120){
504  r = X,g = C,b = 0;
505  }
506  else if(H >= 120 && H < 180){
507  r = 0,g = C,b = X;
508  }
509  else if(H >= 180 && H < 240){
510  r = 0,g = X,b = C;
511  }
512  else if(H >= 240 && H < 300){
513  r = X,g = 0,b = C;
514  }
515  else{
516  r = C,g = 0,b = X;
517  }
518 
519  setRgb((r+m)*255, (g+m)*255, (b+m)*255);
520 }
