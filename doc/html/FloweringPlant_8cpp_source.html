186 // Static method to register a species in the species map, given a species name

192 unsigned int FloweringPlant::registerSpecies(const std::string& species)
193 {
194  unsigned int speciesId = 0;
195 
196  // Do we already have a record of this plant species?
197  auto it = std::find_if( m_sSpeciesMap.begin(),
198  m_sSpeciesMap.end(),
199  [species](const std::pair<unsigned int, std::string> & pair)
200  {return pair.second == species;} );
201  if (it != m_sSpeciesMap.end())
202  {
203  speciesId = it->first;
204  }
205  else
206  {
207  // this is a new species, so select a new species id to use, and also
208  // insert of record of the new id and species name into m_sSpeciesMap
209  speciesId = m_sNextFreeSpeciesId++;
210  m_sSpeciesMap[speciesId] = species;
211 
212  if (ModelParams::verbose())
213  {
214  std::cout << "Adding new plant species to map: id=" << speciesId << ", name=" <<
215  species << std::endl;
216  }
217  }
218 
219  return speciesId;
220 }

331 void FloweringPlant::constructCloggingMap(std::vector<PlantTypeConfig>& ptcs)
332 {
333  assert(m_sCloggingMap.empty());
334 
335  for (PlantTypeConfig& ptc : ptcs)
336  {
337  unsigned int thisSpeciesId = getSpeciesId(ptc.species);
338 
339  if (ptc.pollenCloggingSpecies.empty())
340  {
341  ptc.pollenCloggingAll = false;
342  ptc.pollenCloggingPartial = false;
343 
344  // this species does not clog anything, so add an empty entry
345  // for this species to the clogging map
346  std::vector<unsigned int> cloggingSpIdVec;
347  m_sCloggingMap[thisSpeciesId] = cloggingSpIdVec;
348  }
349  else if (ptc.pollenCloggingSpecies == "all")
350  {
351  ptc.pollenCloggingAll = true;
352  ptc.pollenCloggingPartial = false;
353 
354  // this species clogs everything, so add the ids of all other species to
355  // the entry for this species in the map
356  std::vector<unsigned int> cloggingSpIdVec;
357  for (auto& entry : m_sSpeciesMap)
358  {
359  if (entry.first != thisSpeciesId)
360  {
361  cloggingSpIdVec.push_back(entry.first);
362  }
363  }
364  m_sCloggingMap[thisSpeciesId] = cloggingSpIdVec;
365  }
366  else
367  {
368  ptc.pollenCloggingAll = false;
369  ptc.pollenCloggingPartial = true;
370 
371  // this species clogs some other species, but not necessarily all
372  // so we need to add specific ids to this species' entry in the map
373 
374  // first build a vector of the name of each species that is clogged by this one
375  // (ptc.pollenCloggingSpecies contains the names in a single comma-separated string)
376  std::regex delim(",\\s+"); // comma followed by optional whitespace
377  std::vector<std::string> cloggingSpNameVec {
378  std::sregex_token_iterator(ptc.pollenCloggingSpecies.begin(), ptc.pollenCloggingSpecies.end(), delim, -1),
379  std::sregex_token_iterator()
380  };
381 
382  // now for each species name in the vector, get the corresponding speciesId and add it to
383  // the vector to be used in the clogging map
384  std::vector<unsigned int> cloggingSpIdVec;
385  for (std::string& cloggingSpName : cloggingSpNameVec)
386  {
387  if (ModelParams::verbose())
388  {
389  std::cout << "Species " << ptc.species << " clogs species " << cloggingSpName << std::endl;
390  }
391 
392  cloggingSpIdVec.push_back(getSpeciesId(cloggingSpName));
393  }
394 
395  m_sCloggingMap[thisSpeciesId] = cloggingSpIdVec;
396 
397  }
398  }
399 
400  // Calculate the values of m_sbCloggingAll and m_sbCloggingNone
401  unsigned int numSpecies = m_sCloggingMap.size();
402  unsigned int numEmpty = 0;
403  unsigned int numFull = 0;
404  for (auto& entry : m_sCloggingMap)
405  {
406  if (entry.second.empty())
407  {
408  // this species does not clog any other species
409  ++numEmpty;
410  }
411  else if (entry.second.size() >= numSpecies - 1)
412  {
413  // this species clogs all other species
414  ++numFull;
415  }
416  }
417  m_sbCloggingAll = (numFull == numSpecies);
418  m_sbCloggingNone = (numEmpty == numSpecies);
419 }
