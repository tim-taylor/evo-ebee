64  // taking into account the suggested number from the pollinator, the
65  // remaining capacity of the stigma, and the amount of pollen actually
66  // being carried by the pollinator
67  int attemptedNum = std::min({suggestedNum,
68  m_iStigmaMaxPollenCapacity - (int)m_StigmaPollen.size(),
69  (int)pollinatorStore.size()});
70 
71  int actualNum = 0;
72 
73  // now calculate how many grains we can actually move from the pollinatorStore,
74  // taking account of the pollen species if necessary, and ensure that all grains
75  // to be moved are placed at the end of the pollinatorStore ready for being
76  // moved in bulk
77  if (sameSpeciesOnly)
78  {
79  // We can only transfer pollen of the same species as this flower, so we need
80  // to look through the pollinator's store and select which grains we will take.
81  // For efficiency, we do this by first searching through the pollinator's store
82  // and moving any matching pollen to the end of the store (we exit this process
83  // when we've found enough pollen, or when we've reached the end of the store).
84  if (!pollinatorStore.empty())
85  {
86  auto nextItr = pollinatorStore.begin();
87  auto swapItr = pollinatorStore.end();
88 
89  while ((nextItr != swapItr) && (actualNum < attemptedNum))
90  {
91  if (nextItr->pSource->getSpecies() == getSpecies())
92  {
93  --swapItr;
94  if (nextItr != swapItr)
95  {
96  std::iter_swap(nextItr, swapItr);
97  }
98  ++actualNum;
99  }
100  else
101  {
102  ++nextItr;
103  }
104  }
105  }
106  }
107  else
108  {
109  // No restriction on pollen species, so we can grab any pollen from
110  // the pollinator. Note that the pollinator's store should already be shuffled,
111  // so we can just pick the required number of grains from the end of the
112  // store, and this will be a random selection
113  actualNum = attemptedNum;
114  }
115 
116  // Whether sameSpeciesOnly is true or false, we have now figured out how many grains
117  // are to moved, and the grains we want to move are now at the end of the pollinatorStore.
118  // Now we actually move those grains to the stigma store in one operation, and delete
119  // them from the pollinatorStore.
120  if (actualNum > 0)
121  {
122  std::move(pollinatorStore.end()-actualNum,
123  pollinatorStore.end(),
