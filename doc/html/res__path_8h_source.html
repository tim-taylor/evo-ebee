8 /*
9  * Get the resource path for resources located in res/subDir
10  * It's assumed the project directory is structured like:
11  * bin/
12  * the executable
13  * res/
14  * Lesson1/
15  * Lesson2/
16  *
17  * Paths returned will be Lessons/res/subDir
18  */
19 std::string getResourcePath(const std::string &subDir = ""){
20  //We need to choose the path separator properly based on which
21  //platform we're running on, since Windows uses a different
22  //separator than most systems
23 #ifdef _WIN32
24  const char PATH_SEP = '\\';
25 #else
26  const char PATH_SEP = '/';
27 #endif
28  //This will hold the base resource path: Lessons/res/
29  //We give it static lifetime so that we'll only need to call
30  //SDL_GetBasePath once to get the executable path
31  static std::string baseRes;
32  if (baseRes.empty()){
33  //SDL_GetBasePath will return NULL if something went wrong in retrieving the path
34  char *basePath = SDL_GetBasePath();
35  if (basePath){
36  //baseRes = basePath;
37  baseRes = basePath + std::string("/../"); // [[ Modified by TT ]]
38  SDL_free(basePath);
39  }
40  else {
41  std::cerr << "Error getting resource path: " << SDL_GetError() << std::endl;
42  return "";
43  }
44  //We replace the last bin/ with res/ to get the the resource path
45  size_t pos = baseRes.rfind("bin");
46  baseRes = baseRes.substr(0, pos) + "res" + PATH_SEP;
47  }
48  //If we want a specific subdirectory path in the resource directory
49  //append it to the base path. This would be something like Lessons/res/Lesson0
50  return subDir.empty() ? baseRes : baseRes + subDir + PATH_SEP;
51 }
