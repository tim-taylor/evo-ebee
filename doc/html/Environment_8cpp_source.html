98 void Environment::initialisePlants()
99 {
100  const std::vector<PlantTypeDistributionConfig>& pdconfigs =
101  ModelParams::getPlantTypeDistributionConfigs();
102 
103  for (const PlantTypeDistributionConfig& pdcfg : pdconfigs)
104  {
105  // first find the corresponding PlantTypeConfig
106  const PlantTypeConfig* pPTC = ModelParams::getPlantTypeConfig(pdcfg.species);
107  if (pPTC == nullptr)
108  {
109  throw std::runtime_error("Unknown plant species '" + pdcfg.species +
110  "' specified in config file");
111  }
112 
113  // calculate some basic values associated with the requested distribution
114  int w = std::max(1, 1 + pdcfg.areaBottomRight.x - pdcfg.areaTopLeft.x);
115  int h = std::max(1, 1 + pdcfg.areaBottomRight.y - pdcfg.areaTopLeft.y);
116  int area = w * h;
117  int numPlants = (int)((float)area * pdcfg.density);
118  std::uniform_real_distribution<float> distW(0.0, w);
119  std::uniform_real_distribution<float> distH(0.0, h);
120 
121  // create a 2D array that will record, for each patch in the distribution area,
122  // the coordinates of each plant to be added to that patch
123  std::vector<fPos> patchInfo[w][h];
124 
125  for (int i = 0; i < numPlants; ++i)
126  {
127  fPos fpos{ pdcfg.areaTopLeft.x + distW(EvoBeeModel::m_sRngEngine),
128  pdcfg.areaTopLeft.y + distH(EvoBeeModel::m_sRngEngine) };
129  int iLocalX = (int)fpos.x - pdcfg.areaTopLeft.x;
130  int iLocalY = (int)fpos.y - pdcfg.areaTopLeft.y;
131  patchInfo[iLocalX][iLocalY].push_back(fpos);
132  }
133 
134  // now go through the patchInfo array, and for each patch, go through all plants
135  // to be added to that patch
136  for (int x = 0; x < w; ++x)
137  {
138  for (int y = 0; y < h; ++y)
139  {
140  Patch& patch = getPatch(x + pdcfg.areaTopLeft.x, y + pdcfg.areaTopLeft.y);
141  std::vector<fPos>& posvec = patchInfo[x][y];
142  for (fPos& pos : posvec)
143  {
144  patch.addPlant(*pPTC, pos);
145  }
146 
147  if (pdcfg.refuge || pdcfg.seedOutflowRestricted || (!pdcfg.seedOutflowAllowed))
148  {
149  patch.setReproConstraints(pdcfg);
150  }
151  }
152  }
153  }
154 }

242 void Environment::initialiseNewGeneration()
243 {
244  // generate a new populations of plants based upon pollinated plants from
245  // the previous generation, then delete all plants from previous generation
246  //
248  //
249  /*
250  overall goal:
251  replace all plants (stored in individual Patches) with a
252  new generation of plants
253  assumptions:
254  If PTD area is refuge or seedOutflow is restricted (outflow prob < 1.0), that
255  area is not allowed to overlap with any other area
256  (see implementation details below)
257  constraints:
258  plant type distributions constraints:
259  refuges (restriction on incoming alien speices) (isRefuge/prob)
260  seed outflow (allowed/prob)
261  which plants reproduce?
262  flower is pollinated
263  possibilities:
264  max limits defined by:
265  fixed number of plants in each PTD area (defined by density param)
266  fixed number across whole environment
267  approach:
268  create a vector of all pollinated plants and shuffle it
269  create an empty vector to store new generation
270  for each plant in vector (up to global max repro num for env)
271  consider a nearby position in which to reproduce
272  determine prob that it will successfully reproduce in that position, taking into acccount
273  seed outflow prob of current patch
274  seed inflow prob (refuge incoming) of destination
275  any local PTDConfig density constraints (stored in ModelParams)
276  if successful, create new plant and put in newgenvec
277  empty plant vectors for all patches
278  for each plant in newgenvec, move plant to appropriate Patch vec
279  implementation details:
280  each Patch needs
281  outflow prob (and allowed/restricted flags? corresponding to p>0, p<1)
282  isRefuge flag
283  refugeNativeSpecies id
284  alienInflowProb
285  We can deal with constraints on overlapping PTD areas (see assumptions above) by
286  thowing an exception if there is an attempt to set these Patch probs/flags
287  muliple times for any single Patch
288  New params required:
289  Environment -> repro-constraint-max-density-global
290  PTDConfig -> repro-constraint-max-density-area (if isRefuge or outflowProb=0)
291  */
292 
294  // Step 0: Internal book-keeping
295  resetLocalDensityCounts();
296 
298  // Step 1: Construct a new generation of plants
299 
300  // -- Step 1a: create a vector of all pollinated plants and shuffle it
301  std::vector<FloweringPlant*> pollinatedPlantPtrs;
302  for (Patch& p : m_Patches)
303  {
304  if (p.hasFloweringPlants())
305  {
306  PlantVector & fplants = p.getFloweringPlants();
307  for (FloweringPlant & fplant : fplants)
308  {
309  if (fplant.pollinated())
310  {
311  pollinatedPlantPtrs.push_back(&fplant);
312  }
313  }
314  }
315  }
316  std::shuffle(pollinatedPlantPtrs.begin(), pollinatedPlantPtrs.end(), EvoBeeModel::m_sRngEngine);
317 
318  // -- Step 1b: create an empty vector to store new generation
319  std::vector<FloweringPlant> newPlants;
320 
321  // -- Step 1c: for each plant in vector (up to global max repro num for env)
322  unsigned int globalMax = ModelParams::getReproGlobalDensityConstrained() ?
323  ((unsigned int)((float)m_Patches.size() * ModelParams::getReproGlobalDensityMax())) : 0;
324 
325  for (FloweringPlant* pPlant : pollinatedPlantPtrs)
326  {
327  // -- Step 1c.1: consider a nearby position in which to reproduce
330  float distance = EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine);
331  float heading = EvoBeeModel::m_sDirectionDistrib(EvoBeeModel::m_sRngEngine);
332  fPos delta { distance*std::cos(heading), distance*std::sin(heading) };
333  fPos fCurPos { pPlant->getPosition() };
334  fPos fNewPos { fCurPos + delta };
335 
336  iPos iCurPos = getPatchCoordFromFloatPos(fCurPos);
337  iPos iNewPos = getPatchCoordFromFloatPos(fNewPos);
338 
339  // -- Step 1c.2: determine prob that it will successfully reproduce in that position,
340  // taking into acccount:
341  // - seed outflow prob of current patch
342  // - seed inflow prob (refuge incoming) of destination
343  // - any local PTDConfig density constraints (stored in ModelParams)
344  bool bAnyChance = true;
345  float successProb = 1.0;
346 
347  if (iNewPos != iCurPos)
348  {
349  if (!inEnvironment(iNewPos))
350  {
351  // new plant has fallen off the edge of the world!
352  bAnyChance = false;
353  }
354  else
355  {
356  // new plant is still in environment but in a different patch to old one
357  const Patch& curPatch = pPlant->getPatch();
358 
359  // first consider our chances of successfully leaving the current patch
360  if (!curPatch.seedOutflowAllowed())
361  {
362  // no seed outflow is allowed from this patch!
363  bAnyChance = false;
364  }
365  else if (curPatch.seedOutflowRestricted())
366  {
367  // seed outflow is allowed at a restricted rate
368  // (multiply the prob of doing this with our currently calculated prob)
369  successProb *= curPatch.getSeedOutflowProb();
370  }
371 
372  // now consider chances of succesfully moving into the new patch
373  if (bAnyChance)
374  {
375  Patch& newPatch = getPatch(iNewPos);
376  if (newPatch.refuge() && (newPatch.getRefugeNativeSpeciesId() != pPlant->getSpeciesId()))
377  {
378  // trying to move into a refuge for a different plant species
379  // (multiply the prob of doing this with our currently calculated prob)
380  successProb *= newPatch.getRefugeAlienInflowProb();
381  }
382  }
383  }
384  }
385 
386  // having considered local constraints on the source and desitination patches, now consider
387  // meso-scale density constraints on PlantTypeDistribution areas
388  if (bAnyChance && localDensityLimitReached(iNewPos))
389  {
390  bAnyChance = false;
391  }
392 
393  // -- Step 1c.3: if successful, create new plant and put in newPlants vector
394  if (bAnyChance && (EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine) < successProb))
395  {
396  Patch& newPatch = getPatch(iNewPos);
397 
398  // create a mutated version of the parent plant
399  newPlants.emplace_back(pPlant, fNewPos, &newPatch, true);
400 
401  // and also update local density count
402  incrementLocalDensityCount(iNewPos);
403  }
404 
405  // -- Step 1c.4: if we've now reached the global limit on the number of new plants to
406  // produce, stop!
407  if ((globalMax > 0) && (newPlants.size() > globalMax))
408  {
409  break;
410  }
411  }
412 
413  // -- Step 1d: empty plant vectors for all patches
414  for (Patch& p : m_Patches)
415  {
416  p.killAllPlants();
417  }
418 
419  // -- Step 1e: for each plant in newPlants, move it to appropriate Patch vec
420  for (FloweringPlant& plant : newPlants)
421  {
422  plant.getPatch().addPlant(plant);
423  }
424 
426  // Step 2: Reset all pollinators to their initial state
427  for (Pollinator* pPollinator : m_AllPollinators)
428  {
429  pPollinator->reset();
430  }
431  //
432 
434  // Step 3: anything else to do? I don't think so...
436 }
