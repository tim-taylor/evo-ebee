93 void Environment::initialisePlants()
94 {
95  const std::vector<PlantTypeDistributionConfig> &pdconfigs = ModelParams::getPlantTypeDistributionConfigs();
96  for (const PlantTypeDistributionConfig &pdcfg : pdconfigs)
97  {
98  // first find the corresponding PlantTypeConfig
99  const PlantTypeConfig* pPTC = ModelParams::getPlantTypeConfig(pdcfg.species);
100  if (pPTC == nullptr)
101  {
102  throw std::runtime_error("Unknown plant species '" + pdcfg.species + "' specified in config file");
103  }
104 
105  // calculate some basic values associated with the requested distribution
106  int w = std::max(1, 1 + pdcfg.areaBottomRightX - pdcfg.areaTopLeftX);
107  int h = std::max(1, 1 + pdcfg.areaBottomRightY - pdcfg.areaTopLeftY);
108  int area = w * h;
109  int numPlants = (int)((float)area * pdcfg.density);
110  std::uniform_real_distribution<float> distW(0.0, w);
111  std::uniform_real_distribution<float> distH(0.0, h);
112 
113  // create a 2D array that will record, for each patch in the distribution area,
114  // the coordinates of each plant to be added to that patch
115  std::vector<fPos> patchInfo[w][h];
116 
117  for (int i = 0; i < numPlants; ++i)
118  {
119  fPos fpos{ pdcfg.areaTopLeftX + distW(EvoBeeModel::m_sRngEngine),
120  pdcfg.areaTopLeftY + distH(EvoBeeModel::m_sRngEngine) };
121  int iLocalX = (int)fpos.x - pdcfg.areaTopLeftX;
122  int iLocalY = (int)fpos.y - pdcfg.areaTopLeftY;
123  patchInfo[iLocalX][iLocalY].push_back(fpos);
124  }
125 
126  // now go through the patchInfo array, and for each patch, go through all plants
127  // to be added to that patch
128  for (int x = 0; x < w; ++x)
129  {
130  for (int y = 0; y < h; ++y)
131  {
132  Patch &p = getPatch(x + pdcfg.areaTopLeftX, y + pdcfg.areaTopLeftY);
133  auto &v = patchInfo[x][y];
134  for (auto &pinfo : v)
135  {
136  // pinfo.first is x coordinate, second is y coordinate
137  p.addPlant(*pPTC, pinfo);
138  }
139  }
140  }
141  }
142 }
