113 
114  for (const PlantTypeDistributionConfig& pdcfg : pdconfigs)
115  {
116  // first find the corresponding PlantTypeConfig
117  const PlantTypeConfig* pPTC = ModelParams::getPlantTypeConfig(pdcfg.species);
118  if (pPTC == nullptr)
119  {
120  throw std::runtime_error("Unknown plant species '" + pdcfg.species +
121  "' specified in config file");
122  }
123 
124  // calculate some basic values associated with the requested distribution
125  int w = std::max(1, 1 + pdcfg.areaBottomRight.x - pdcfg.areaTopLeft.x);
126  int h = std::max(1, 1 + pdcfg.areaBottomRight.y - pdcfg.areaTopLeft.y);
127  int area = w * h;
128  int numPlants = (int)((float)area * pdcfg.density);
129  std::uniform_real_distribution<float> distW(0.0, w - EvoBee::SMALL_FLOAT_NUMBER);
130  std::uniform_real_distribution<float> distH(0.0, h - EvoBee::SMALL_FLOAT_NUMBER);
131 
132  // create a 2D array that will record, for each patch in the distribution area,
133  // the coordinates of each plant to be added to that patch
134  std::vector<fPos> patchInfo[w][h];
135 
136  for (int i = 0; i < numPlants; ++i)
137  {
138  fPos fpos{ pdcfg.areaTopLeft.x + distW(EvoBeeModel::m_sRngEngine),
139  pdcfg.areaTopLeft.y + distH(EvoBeeModel::m_sRngEngine) };
140  int iLocalX = (int)fpos.x - pdcfg.areaTopLeft.x;
141  int iLocalY = (int)fpos.y - pdcfg.areaTopLeft.y;
142  patchInfo[iLocalX][iLocalY].push_back(fpos);
143  }
144 
145  // now go through the patchInfo array, and for each patch, go through all plants
146  // to be added to that patch
147  for (int x = 0; x < w; ++x)
148  {
149  for (int y = 0; y < h; ++y)
150  {

152  std::vector<fPos>& posvec = patchInfo[x][y];

156  }

213 // (Moore neighbourhood), and return a pointer to the closest plant found,
214 // or nullptr if none found.
215 //
216 // The second parameter (default value = 1.0) specifies a maximum search radius.
217 // If this is given a zero or negative value, then it is ignored, and all flowers
218 // withiin the Moore neighbourhood are considered regardless of distance (but
219 // note that this means that the search radius is effectively asymmetric in
220 // different directions because the patches are squares). If the radius is set
221 // to a positive value, then only flowers within that distance of fpos will be
222 // considered. Therefore, if fRadius is positive, it only really makes sense for
223 // it to have a value of <= 1.0, as above 1.0 the search might be clipped depending
224 // upon the location of fpos within the central patch.

552 void Environment::initialiseNewGeneration()
553 {
554 
555 
557  // Step 0: Internal book-keeping
558  resetLocalDensityCounts();
559 
561  // Step 1: Construct a new generation of plants
562 
563  // -- Step 1a: create a vector of all pollinated plants and shuffle it
564  std::vector<FloweringPlant*> pollinatedPlantPtrs;
565  for (Patch& p : m_Patches)
566  {
567  if (p.hasFloweringPlants())
568  {
569  PlantVector& plants = p.getFloweringPlants();
570  for (FloweringPlant& plant : plants)
571  {
572  if (plant.pollinated())
573  {
574  // for each pollinated plant, we look at each pollen grain on
575  // the stigma of each of its flowers, and add a candidate
576  // pointer for reproduction for each conspecific grain
577  const std::vector<Flower>& flowers = plant.getFlowers();
578  for (const Flower& flower : flowers)
579  {
580  const PollenVector& stigmaPollen = flower.getStigmaPollen();
581  for (const Pollen& pollen : stigmaPollen)
582  {
583  if (pollen.speciesId == flower.getSpeciesId())
584  {
585  pollinatedPlantPtrs.push_back(&plant);
589  }
590  }
591  }
592  }
593  }
594  }
595  }
596  std::shuffle(pollinatedPlantPtrs.begin(), pollinatedPlantPtrs.end(), EvoBeeModel::m_sRngEngine);
597 
598  // -- Step 1b: create an empty vector to store new generation
599  std::vector<FloweringPlant> newPlants;
600 
601  // -- Step 1c: for each plant in vector (up to global max repro num for env)
602  unsigned int globalMax = ModelParams::getReproGlobalDensityConstrained() ?
603  ((unsigned int)((float)m_Patches.size() * ModelParams::getReproGlobalDensityMax())) : 0;
604 
605  for (FloweringPlant* pPlant : pollinatedPlantPtrs)
606  {
607  // -- Step 1c.1: consider a nearby position in which to reproduce
608  const Patch& curPatch = pPlant->getPatch();
609 
610  fPos fCurPos { pPlant->getPosition() };
611  fPos fNewPos;
612 
613  if (pPlant->reproSeedDispersalGlobal())
614  {
615  if (curPatch.seedOutflowAllowed())
616  {
617  // seed outflow from this patch's locality is allowed, so we now need
618  // to decide whether we can pick a destination position from the entire
619  // environment or just from the locality defined by m_ReproRestrictionAreaTL/BR
620  if ((!curPatch.seedOutflowRestricted()) ||
621  (EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine) < curPatch.getSeedOutflowProb()))
622  {
623  fNewPos = getRandomPositionF();
624  }
625  else
626  {
627  fNewPos = getRandomPositionF(curPatch.getReproRestrictionAreaTopLeft(),
628  curPatch.getReproRestrictionAreaBottomRight());
629  }
630  }
631  else
632  {
633  // seed outflow from this patch's locaility is not allowed, so in
634  // this case "SeedDispersalGlobal" means anywhere within the locaility
635  // defined by m_ReproRestrictionAreaTL/BR
636  fNewPos = getRandomPositionF(curPatch.getReproRestrictionAreaTopLeft(),
637  curPatch.getReproRestrictionAreaBottomRight());
638  }
639  }
640  else
641  {
642  // pick position from within a specified radius of the current position
643  std::uniform_real_distribution<float> distanceDistrib(0.0, pPlant->reproSeedDispersalRadius());
644  float distance = distanceDistrib(EvoBeeModel::m_sRngEngine);
645  float heading = EvoBeeModel::m_sDirectionDistrib(EvoBeeModel::m_sRngEngine);
646  fPos delta { distance*std::cos(heading), distance*std::sin(heading) };
647  fNewPos = fCurPos + delta;
648  }
649 
650  iPos iCurPos = getPatchCoordFromFloatPos(fCurPos);
651  iPos iNewPos = getPatchCoordFromFloatPos(fNewPos);
652 
653  // -- Step 1c.2: determine prob that it will successfully reproduce in that position,
654  // taking into acccount:
655  // - seed outflow prob of current patch
656  // - seed inflow prob (refuge incoming) of destination
657  // - any local PTDConfig density constraints (stored in ModelParams)
658  bool bAnyChance = true;
659  float successProb = 1.0;
660 
661  if (iNewPos != iCurPos)
662  {
663  if (!inEnvironment(iNewPos))
664  {
665  // new plant has fallen off the edge of the world!
666  bAnyChance = false;
667  }
668  else
669  {
670  // new plant is still in environment but in a different patch to old one.
671  // consider chances of succesfully moving into the new patch
672  Patch& newPatch = getPatch(iNewPos);
673  if (newPatch.refuge() && (newPatch.getRefugeNativeSpeciesId() != pPlant->getSpeciesId()))
674  {
675  // trying to move into a refuge for a different plant species
676  // (multiply the prob of doing this with our currently calculated prob)
677  successProb *= newPatch.getRefugeAlienInflowProb();
678  }
679  }
680  }
681 
682  // having considered local constraints on the source and desitination patches, now consider
683  // meso-scale density constraints on PlantTypeDistribution areas
684  if (bAnyChance && localDensityLimitReached(iNewPos))
685  {
686  bAnyChance = false;
687  }
688 
689  // -- Step 1c.3: if successful, create new plant and put in newPlants vector
690  if (bAnyChance && (EvoBeeModel::m_sUniformProbDistrib(EvoBeeModel::m_sRngEngine) < successProb))
691  {
692  Patch& newPatch = getPatch(iNewPos);
693 
694  // create a mutated version of the parent plant
695  newPlants.emplace_back(pPlant, fNewPos, &newPatch, true);
696 
697  // and also update local density count
698  incrementLocalDensityCount(iNewPos);
699  }
700 
701  // -- Step 1c.4: if we've now reached the global limit on the number of new plants to
702  // produce, stop!
703  if ((globalMax > 0) && (newPlants.size() > globalMax))
704  {
705  break;
706  }
707  }
708 
709  // -- Step 1d: empty plant vectors for all patches
710  for (Patch& p : m_Patches)
711  {
712  p.killAllPlants();
713  }
714 
715  // -- Step 1e: for each plant in newPlants, move it to appropriate Patch vec
