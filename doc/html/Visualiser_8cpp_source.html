169  150, 150, 150, 255

192  boxRGBA(

194  getScreenCoordFromFloatX(hivepos.x),
195  getScreenCoordFromFloatY(hivepos.y),
196  getScreenCoordFromFloatWithAbsOffsetX(hivepos.x+1.0, -1),
197  getScreenCoordFromFloatWithAbsOffsetY(hivepos.y+1.0, -1),

220  unsigned int maxid = Pollinator::getMaxIdIssued();
221  Uint8 r = 100 + ((150 * id) / maxid);
222  Uint8 g = 250 - ((150 * id) / maxid);
223  Uint8 b = r;
224

226  auto itr = m_PollMoveHists.find(id);

242  if (posHist.size() > 1)
