226 // Collect pollen from flower if available, and add to our store
227 // If any pollen is added, the contents of the store is shuffled so that
228 // subsequent operations that act of randomly chosen grains from the store
229 // can simply and efficiently pick off grains from the end of the store.
230 // If the store's maxium capacity is exceeded, some random grains are
231 // removed to bring it down to the maximum allowed size.
232 //
233 void Pollinator::collectPollenFromAnther(Flower* pFlower)
234 {
235  // transfer anther pollen from flower to our store
236  int num = pFlower->transferAntherPollenToPollinator(m_PollenStore);
237 
238  // if some new grains have been added, shuffle the whole store now
239  // so that subsequent removals of pollen happen in a random order
240  if (num > 0)
241  {
242  std::shuffle(m_PollenStore.begin(), m_PollenStore.end(), EvoBeeModel::m_sRngEngine);
243  }
244 
245  // finally, if the number of grains in our store now exceeds the
246  // maximum capacity, delete some. As we've just shuffled the store,
247  // we can just remove the required number from the end of the store
248  int xs = m_PollenStore.size() - m_iMaxPollenCapacity;
249  if (xs > 0)
250  {
251  m_PollenStore.erase(m_PollenStore.end()-xs, m_PollenStore.end());
252  }
253 }
