147 // Collect pollen from flower if available, and add to our store
148 void Pollinator::collectPollenFromAnther(Flower* pFlower)
149 {
150  // transfer anther pollen from flower to our store
151  int num = pFlower->transferAntherPollenToPollinator(m_PollenStore);
152 
153  // if some new grains have been added, shuffle the whole store now
154  // so that subsequent removals of pollen happen in a random order
155  if (num > 0)
156  {
157  std::shuffle(m_PollenStore.begin(), m_PollenStore.end(), EvoBeeModel::m_sRngEngine);
158  }
159 
160  // finally, if the number of grains in our store now exceeds the
161  // maximum capacity, delete some. As we've just shuffled the store,
162  // we can just remove the required number from the end of the store
163  int xs = m_PollenStore.size() - m_iMaxPollenCapacity;
164  if (xs > 0)
165  {
166  m_PollenStore.erase(m_PollenStore.end()-xs, m_PollenStore.end());
167  }
168 }
